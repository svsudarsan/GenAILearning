// createLead.single.spec.js
// Playwright single-test suite for CRM application (Create Lead - happy path)
// - Single test (as requested).
// - XPath-only locators (prefer relative XPath).
// - Uses explicit waits (waitForSelector / waitForNavigation). NO thread.sleep.
// - Login uses <form id="login"> and credentials demosalesmanager / crmsfa.
// - Navigates: click CRM/SFA -> Leads -> Create Lead.
// - Uses the <tbody> container to scope lead input actions.
// - Uses ID locators when they are meaningful and do not contain numeric-only ids (e.g. createLeadForm_companyName).
// - Does NOT use CSS selectors.

const { test, expect } = require('@playwright/test');

test.describe('CRM - Create Lead (single test)', () => {
  const BASE_URL = 'http://leaftaps.com/opentaps/control/main'; // <<-- set your app URL

  /**
   * Perform login using the form with id="login"
   * - fills username / password and submits.
   */
  async function loginUsingLoginForm(page) {
    // Navigate to app (login page)
    await page.goto(BASE_URL);

    // Wait for the login form to be visible
    await page.waitForSelector("xpath=//form[@id='login']", { state: 'visible' });

    // Fill username (inside the login form)
    await page.locator("xpath=//form[@id='login']//input[@id='username' or @name='USERNAME']").fill('demosalesmanager');

    // Fill password
    await page.locator("xpath=//form[@id='login']//input[@id='password' or @name='PASSWORD']").fill('crmsfa');

    // Click the submit button inside the form and wait for navigation
    await Promise.all([
      page.waitForNavigation({ waitUntil: 'networkidle' }),
      page.locator("xpath=//form[@id='login']//input[@type='submit' and (@value='Login' or contains(@class,'decorativeSubmit'))]").click()
    ]);
  }

  /**
   * Navigate to Create Lead page: Click CRM/SFA -> Leads -> Create Lead
   * All clicks use XPath and explicit navigation waits.
   */
  async function openCreateLeadPage(page) {
    // Click the CRM/SFA link (text may be 'CRM/SFA' or contain 'CRM')
    const crmLinkXpath = "xpath=//a[normalize-space(text())='CRM/SFA' or contains(translate(normalize-space(.),'ABCDEFGHIJKLMNOPQRSTUVWXYZ','abcdefghijklmnopqrstuvwxyz'),'crm')]";
    await page.waitForSelector(crmLinkXpath, { state: 'visible' });
    await Promise.all([
      page.waitForNavigation({ waitUntil: 'networkidle' }),
      page.locator(crmLinkXpath).click()
    ]);

    // Click Leads link
    const leadsLinkXpath = "xpath=//a[normalize-space(text())='Leads' or contains(normalize-space(.),'Leads')]";
    await page.waitForSelector(leadsLinkXpath, { state: 'visible' });
    await Promise.all([
      page.waitForNavigation({ waitUntil: 'networkidle' }),
      page.locator(leadsLinkXpath).click()
    ]);

    // Click Create Lead link (could be 'Create Lead' or 'Create Leads')
    const createLeadLinkXpath = "xpath=//a[normalize-space(text())='Create Lead' or normalize-space(text())='Create Leads' or contains(normalize-space(.),'Create Lead')]";
    await page.waitForSelector(createLeadLinkXpath, { state: 'visible' });
    await Promise.all([
      page.waitForNavigation({ waitUntil: 'networkidle' }),
      page.locator(createLeadLinkXpath).click()
    ]);
  }

  /**
   * The single test: fills mandatory fields inside the <tbody> and submits the form.
   * Mandatory inputs (per provided DOM):
   *  - companyName (id=createLeadForm_companyName, name=companyName)
   *  - firstName  (id=createLeadForm_firstName, name=firstName)
   *  - lastName   (id=createLeadForm_lastName, name=lastName)
   */
  test('P1 - Create Lead (happy path) - create lead with mandatory fields', async ({ page }) => {
    // 1) Login
    await loginUsingLoginForm(page);

    // 2) Open Create Lead page
    await openCreateLeadPage(page);

    // 3) Wait for tbody (form fields area) to be present and visible
    const tbodyXpath = "xpath=//tbody";
    await page.waitForSelector(tbodyXpath, { state: 'visible' });

    // 4) Fill Company Name - prefer ID when it's meaningful (createLeadForm_companyName)
    const companyInput = page.locator(`${tbodyXpath}//input[@id='createLeadForm_companyName' or @name='companyName']`);
    await companyInput.waitFor({ state: 'visible' });
    await companyInput.fill('Testleaf');

    // Verify the input value pre-submit (explicit check)
    await expect(companyInput).toHaveValue('Testleaf');

    // 5) Fill First Name
    const firstNameInput = page.locator(`${tbodyXpath}//input[@id='createLeadForm_firstName' or @name='firstName']`);
    await firstNameInput.waitFor({ state: 'visible' });
    await firstNameInput.fill('Sudarsan');

    // 6) Fill Last Name
    const lastNameInput = page.locator(`${tbodyXpath}//input[@id='createLeadForm_lastName' or @name='lastName']`);
    await lastNameInput.waitFor({ state: 'visible' });
    await lastNameInput.fill('s');

    // 7) Click Create Lead submit button (submit input inside the tbody)
    const createButton = page.locator(`${tbodyXpath}//input[@type='submit' and (@name='submitButton' or @value='Create Lead')]`);
    await createButton.waitFor({ state: 'visible' });

    // Submit and wait for navigation to the result/view page
    await Promise.all([
      page.waitForNavigation({ waitUntil: 'networkidle' }),
      createButton.click()
    ]);

    // 8) Post-submit verification:
    // Primary check: span with id viewLead_companyName_sp (common in this app)
    const viewCompanySpanXpath = "xpath=//span[@id='viewLead_companyName_sp']";
    if (await page.locator(viewCompanySpanXpath).count() > 0) {
      const span = page.locator(viewCompanySpanXpath);
      await span.waitFor({ state: 'visible' });
      await expect(span).toContainText('Testleaf', { timeout: 5000 });
      return;
    }

    // Fallback: find the table cell labelled 'Company Name' and check adjacent td
    const companyCellFallbackXpath = "xpath=(//td[normalize-space(.)='Company Name' or contains(normalize-space(.),'Company Name')]/following-sibling::td)[1]";
    await page.waitForSelector(companyCellFallbackXpath, { state: 'visible' });
    const companyCell = page.locator(companyCellFallbackXpath);
    await expect(companyCell).toContainText('Testleaf', { timeout: 5000 });
  });
